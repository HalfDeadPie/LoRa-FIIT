30// We use some for headers, keeping fewer for RadioHead messages
31#define RH_RF95_MAX_PAYLOAD_LEN RH_RF95_FIFO_SIZE

33// The length of the headers we add.

37// This is the maximum message length that can be supported by this driver.
38// Can be pre-defined to a smaller size (to save SRAM) prior to including this header
39// Here we allow for 1 byte message length, 4 bytes headers, user data and 2 bytes of FCS
